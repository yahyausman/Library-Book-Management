Database Schema

MySQL table structure for the books system

books Table

Run this in MySQL Workbench

CREATE DATABASE library_db;
USE library_db;

CREATE TABLE books (
  book_id      INT AUTO_INCREMENT PRIMARY KEY,
  title        VARCHAR(255) NOT NULL,
  author       VARCHAR(150) NOT NULL,
  category     VARCHAR(100),
  isbn         VARCHAR(20) UNIQUE,
  publisher    VARCHAR(150),
  year         INT,
  quantity     INT DEFAULT 0,
  available    INT DEFAULT 0,
  created_at   TIMESTAMP DEFAULT NOW()
);

-- Flask routes connect to this table directly
-- Add: INSERT INTO books (title, author, ...) VALUES (%s, %s, ...)
-- View: SELECT * FROM books
-- Search: SELECT * FROM books WHERE title LIKE '%query%'
-- Update: UPDATE books SET title=%s, ... WHERE book_id=%s
-- Delete: DELETE FROM books WHERE book_id=%s
